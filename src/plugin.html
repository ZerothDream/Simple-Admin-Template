固定盒子

  • HTML
  • SASS
  • JS

                            

                            

// 引入plugin.js中的 fixBox 插件方法，并进行调用
var cancel;
$('.wrapper').fixBox({
    selector: '#box-fix',
    getTop: function($box) {
        return $('.navbar').height() + 15;
    },
    getScrollTopLimit: function($box) {
        // 参考的对象 —— 这里相对于“面包屑”
        $refer = $('.breadcrumb');
        return $refer.offset().top - $('.navbar').height() + $refer.outerHeight();
    },
    getWidth: function($box) {
        return $('#box-refer').outerWidth();
    },
    cancel: function(fn) {
        cancel = fn;
    }
});
// 调用 cancel 方法取消固定
if(cancel) {
    cancel();
}
